# This is the Python code running the logic behind the UI above

import random   # Module that gives us random numbers

def give_hint(guess, secret_number):
    difference = abs(guess - secret_number)  # How far away?

    if guess == secret_number:
        return "correct"
    elif guess < secret_number:
        direction = "Too low ⬆"
    else:
        direction = "Too high ⬇"

    if difference <= 5:
        temperature = "🔥 Burning hot!"
    elif difference <= 15:
        temperature = "☀️ Getting warm..."
    elif difference <= 30:
        temperature = "❄️ Cold..."
    else:
        temperature = "🧊 Freezing!"

    return f"{direction} — {temperature}"

def play_game(max_attempts):
    secret = random.randint(1, 100)  # Pick a hidden number
    history = []                       # Empty list for guesses
    attempts = 0

    while attempts < max_attempts:    # Loop until attempts run out
        guess = int(input("Your guess: "))
        attempts += 1
        history.append(guess)          # Add to the list

        result = give_hint(guess, secret)
        if result == "correct":
            print(f"🎉 You got it in {attempts} tries!")
            break
        print(result)

import random

    return random.randint(1, 6)   # One random number 1–6

def roll_multiple_dice(num_dice):
    results = []                      # Start with empty list
    for i in range(num_dice):        # Loop once per die
        result = roll_single_die()
        results.append(result)        # Add to list
    return results

def calculate_total(dice_results):
    total = 0
    for value in dice_results:       # Add up each die value
        total = total + value
    return total

def determine_winner(p1_name, p1_score, p2_name, p2_score):
    if p1_score > p2_score:
        return f"{p1_name} wins! 🏆"
    elif p2_score > p1_score:
        return f"{p2_name} wins! 🏆"
    else:
        return "It's a tie! 🤝"
